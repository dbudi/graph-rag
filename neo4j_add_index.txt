# Vector Index (WAJIB untuk GraphRAG)
CREATE VECTOR INDEX vector
FOR (c:Chunk)
ON (c.embedding)
OPTIONS {
  indexConfig: {
    `vector.dimensions`: 1536,
    `vector.similarity_function`: 'cosine'
  }
};
